584 template <class Rep = long, class Period = std::ratio<1>>
585 bool joinMotionUnsafe(
586 std::unique_lock<std::mutex> &lock,
587 const std::optional<std::chrono::duration<Rep, Period>> &timeout = std::nullopt) {
588 while (motion_generator_running_) {
589 if (timeout.has_value()) {
590 if (control_finished_condition_.wait_for(lock, timeout.value()) == std::cv_status::timeout) {
591 return false;
592 }
593 } else {
594 control_finished_condition_.wait(lock);
595 }
596 }
597 if (control_thread_.joinable()) control_thread_.join();
598 if (control_exception_ != nullptr) {
599 auto control_exception = control_exception_;
600 control_exception_ = nullptr;
601 std::rethrow_exception(control_exception);
602 }
603 return true;
604 }

606 template <typename ControlSignalType>
607 void moveInternal(
608 const std::shared_ptr<Motion<ControlSignalType>> &motion,
609 const std::function<void(const ControlFunc<ControlSignalType> &)> &control_func_executor, bool async) {
610 if (motion == nullptr) {
611 throw std::invalid_argument("The motion must not be null.");
612 }
613 // Fail-fast check in the user thread; the authoritative lock-free check is in Motion::init.
614 if (motion->has_started()) {
615 throw MotionReuseException(
616 "This motion object has already been started before. Motions cannot be reused; create a new motion "
617 "instance instead.");
618 }
619 { // Do not remove brace, it is needed to scope the lock
620 std::unique_lock lock(*control_mutex_);
621 if (is_in_control_unsafe() && motion_generator_running_) {
622 if (!std::holds_alternative<MotionGenerator<ControlSignalType>>(motion_generator_)) {
623 throw InvalidMotionTypeException(
624 "The type of motion cannot change during runtime. Please ensure "
625 "that the "
626 "previous motion finished before using a new type of motion.");
627 }
628 std::get<MotionGenerator<ControlSignalType>>(motion_generator_).updateMotion(motion);
629 } else {
630 joinMotionUnsafe(lock);
631
632 motion_generator_.emplace<MotionGenerator<ControlSignalType>>(this, motion);
633 auto motion_generator = &std::get<MotionGenerator<ControlSignalType>>(motion_generator_);
634 motion_generator->registerUpdateCallback(
635 [this](const RobotState &robot_state, franka::Duration duration, franka::Duration time) {
636 state_buffer_.set(robot_state);
637 });
638 motion_generator_running_ = true;
639 control_thread_ = std::thread([this, control_func_executor, motion_generator]() {
640 try {
641 bool done = false;
642 RobotStateEstimator robot_state_estimator(
643 params_.kalman_q_process_var,
644 params_.kalman_dq_process_var,
645 params_.kalman_ddq_process_var,
646 params_.kalman_control_process_var,
647 params_.kalman_q_obs_var,
648 params_.kalman_dq_obs_var,
649 params_.kalman_q_d_obs_var,
650 params_.kalman_dq_d_obs_var,
651 params_.kalman_ddq_d_obs_var,
652 params_.kalman_control_adaptation_rate);
653 while (!done) {
654 control_func_executor(
655 [this, motion_generator, &robot_state_estimator](const franka::RobotState &rs, franka::Duration d) {
656 return (*motion_generator)(robot_state_estimator.update(rs, *model_), d);
657 });
658 std::unique_lock lock(*control_mutex_);
659
660 // This code is just for the case that a new motion is set just
661 // after the old one terminates. If this happens, we need to
662 // continue with this motion, unless an exception occurs.
663 done = !motion_generator->has_new_motion();
664 if (motion_generator->has_new_motion()) {
665 motion_generator->resetTimeUnsafe();
666 } else {
667 done = true;
668 motion_generator_running_ = false;
669 control_finished_condition_.notify_all();
670 }
671 }
672 } catch (...) {
673 std::unique_lock lock(*control_mutex_);
674 control_exception_ = std::current_exception();
675 motion_generator_running_ = false;
676 control_finished_condition_.notify_all();
677 }
678 });
679 }
680 }
681 if (!async) joinMotion();
682 }
