165inline Vector7d computeJointLimitTorque(
166 const Vector7d &q, const Vector7d &dq, const Vector7d &lower_joint_limits, const Vector7d &upper_joint_limits,
167 double joint_limit_activation_distance, double joint_limit_stiffness, double joint_limit_damping,
168 double joint_limit_max_torque) {
169 Vector7d tau_limit = Vector7d::Zero();
170 const double activation_distance = std::max(joint_limit_activation_distance, 1e-6);
171
172 for (size_t i = 0; i < 7; i++) {
173 const double lower_soft_limit = lower_joint_limits[i] + activation_distance;
174 const double upper_soft_limit = upper_joint_limits[i] - activation_distance;
175
176 if (q[i] < lower_soft_limit) {
177 const double penetration = (lower_soft_limit - q[i]) / activation_distance;
178 double tau = joint_limit_stiffness * (std::exp(penetration) - 1.0);
179 if (dq[i] < 0.0) tau += joint_limit_damping * (-dq[i]);
180 tau_limit[i] = std::min(tau, joint_limit_max_torque);
181 } else if (q[i] > upper_soft_limit) {
182 const double penetration = (q[i] - upper_soft_limit) / activation_distance;
183 double tau = -joint_limit_stiffness * (std::exp(penetration) - 1.0);
184 if (dq[i] > 0.0) tau -= joint_limit_damping * dq[i];
185 tau_limit[i] = std::max(tau, -joint_limit_max_torque);
186 }
187 }
188
189 return tau_limit;
190}

192inline Vector7d computeFrictionCompensation(const Vector7d &dq, const FrictionCompensationParams &params) {
193 Vector7d tau = Vector7d::Zero();
194 const double epsilon = std::max(params.velocity_epsilon, 1e-9);
195 for (int i = 0; i < 7; ++i) {
196 const double limit = std::max(params.max_torque[i], 0.0);
197 const double uncompensated = params.coulomb[i] * std::tanh(dq[i] / epsilon) + params.viscous[i] * dq[i];
198 tau[i] = std::clamp(uncompensated, -limit, limit);
199 }
200 return tau;
201}
